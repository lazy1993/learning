const fibonacci = function fibonacci(n) {
  if (typeof n !== 'number') {
    throw new Error('n 必须为数字！');
  }
  if (n > 10) {
    throw new Error('n 必须小于10');
  }
  if (n < 0) {
    throw new Error('n 必须大于0');
  }
 
  if (n < 2) {
    return n;
  }
 
  return fibonacci(n-1) + fibonacci(n-2);
}
